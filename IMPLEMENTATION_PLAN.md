```diff
*** /dev/null
--- b/scripts/ci/check_secret_refs.py
@@
+#!/usr/bin/env python3
+import sys
+import yaml
+
+ALLOWLIST = {
+    "flux-system",
+    "sops-age",
+    "wireguard-client-conf",
+}
+
+def walk(node):
+    if isinstance(node, dict):
+        yield node
+        for v in node.values():
+            yield from walk(v)
+    elif isinstance(node, list):
+        for i in node:
+            yield from walk(i)
+
+def main(path):
+    declared = set()
+    refs = set()
+    with open(path, "r", encoding="utf-8") as f:
+        docs = list(yaml.safe_load_all(f))
+    for d in docs:
+        if not isinstance(d, dict):
+            continue
+        if d.get("kind") == "Secret":
+            name = ((d.get("metadata") or {}).get("name"))
+            if name:
+                declared.add(name)
+        for n in walk(d):
+            if not isinstance(n, dict):
+                continue
+            skr = n.get("secretKeyRef")
+            if isinstance(skr, dict) and skr.get("name"):
+                refs.add(skr["name"])
+            sr = n.get("secretRef")
+            if isinstance(sr, dict) and sr.get("name"):
+                refs.add(sr["name"])
+            sn = n.get("secretName")
+            if isinstance(sn, str):
+                refs.add(sn)
+    missing = sorted(x for x in refs if x not in declared and x not in ALLOWLIST)
+    if missing:
+        print("Unresolved secret refs:")
+        for m in missing:
+            print(f"- {m}")
+        return 1
+    print("All secret references resolved (after allowlist).")
+    return 0
+
+if __name__ == "__main__":
+    if len(sys.argv) != 2:
+        print("usage: check_secret_refs.py <rendered.yaml>")
+        sys.exit(2)
+    sys.exit(main(sys.argv[1]))
```
